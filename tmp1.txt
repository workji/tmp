SELECT 
  doc.BRANCH_CODE AS BRANCH_CODE,
  doc.SHOUHIN_NO_ORG AS SHOUHIN_NO_ORG,
  search.MUFG_BRANCH_CODE AS MUFG_BRANCH_CODE,
  search.CONTR_INDV_FAMILY_NAME_KANA AS CONTR_INDV_FAMILY_NAME_KANA,
  search.CONTR_INDV_GIVEN_NAME_KANA AS CONTR_INDV_GIVEN_NAME_KANA,
  regd.EMPLOYEE_NO AS EMPLOYEE_NO,
  doc.PRODUCT_ID AS PRODUCT_ID,
  doc.SHUMOKU_CD AS SHUMOKU_CD,
  to_char(doc.CREATE_YMD_DATE, 'YYYY-MM-DD') AS CREATE_YMD_DATE,
  to_char(search.MUFG_CREATE_YMD_DATE, 'YYYY-MM-DD') AS MUFG_CREATE_YMD_DATE,
  to_char(search.UPDATE_YMD_DATE, 'YYYY-MM-DD') AS UPDATE_YMD_DATE,
  to_char(search.MUFG_UPDATE_YMD_DATE, 'YYYY-MM-DD') AS MUFG_UPDATE_YMD_DATE,
  doc.DESIGN_DOC_NO AS DESIGN_DOC_NO,
  doc.DESIGN_DOC_STATUS AS DESIGN_DOC_STATUS,
  doc.DESIGN_DOC_ID AS DESIGN_DOC_ID,
  doc.DESIGN_ID AS DESIGN_ID,
  sm.SHUMOKU_NAME AS SHUMOKU_NAME,
  sa.PET_NAME AS PET_NAME,
  search.CALC_BASE_DATE AS CALC_BASE_DATE,
  doc.CONTENTS AS CONTENTS,
  agency.DAIRITEN_ID AS DAIRITEN_ID,
  agency.DAIRITEN_NAME AS DAIRITEN_NAME,
  app.PROC_STATUS AS PROC_STATUS,
  app.APP_NO AS APP_NO,
  app.APP_ID AS APP_ID,
  usr.CREATOR AS CREATOR,
  doc.DOMAIN_LAST_MOD_ON AS DOMAIN_LAST_MOD_ON,
  doc.EXPIRED AS EXPIRED,
  mn.MGMT_NO AS MGMT_NO,
  ... -- 其他字段（如 kanji name 等）
FROM T_D_DESIGN_DOC doc
INNER JOIN T_D_DESIGN_DOC_FOR_SEARCH search 
  ON search.DESIGN_DOC_ID = doc.DESIGN_DOC_ID
INNER JOIN USER_INFO usr 
  ON usr.USER_ID = doc.CREATOR
INNER JOIN T_S_AGENT_REGD_INFO regd 
  ON regd.USER_ID = doc.CREATOR
INNER JOIN SHUMOKU_AGENCY sa 
  ON sa.SHUMOKU_CD = doc.SHUMOKU_CD 
  AND sa.DAIRITEN_ID = doc.DAIRITEN_ID
INNER JOIN HW_AGENCY agency 
  ON agency.DAIRITEN_ID = doc.DAIRITEN_ID
LEFT JOIN T_S_APP app 
  ON app.DESIGN_DOC_NO = doc.DESIGN_DOC_NO
LEFT JOIN USER_INFO appusr 
  ON appusr.USER_ID = app.CREATOR
LEFT JOIN T_S_AGENT_REGD_INFO appregd 
  ON appregd.USER_ID = app.CREATOR
LEFT JOIN T_S_MGMT_NO mn 
  ON mn.DESIGN_DOC_NO = doc.DESIGN_DOC_NO
-- 注意：此处还有 CASE WHEN 逻辑处理 MUFG/非MUFG 字段优先级（见原代码行 158-200）
