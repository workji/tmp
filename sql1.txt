SELECT 
    taApp.APP_ID AS APP_ID, 
    tsOrderID.CREDIT_REGISTRATION_STATUS AS CREDIT_REGISTRATION_STATUS, 
    taApp.DOMAIN_LAST_MOD_ON AS DOMAIN_LAST_MOD_ON
FROM 
    T_A_APP taApp
INNER JOIN 
    T_S_ORDER_ID tsOrderID ON taApp.APP_NO = tsOrderID.APP_NO
WHERE 
    tsOrderID.CREDIT_REGISTRATION_STATUS = ?  -- 参数1: 信用卡登记状态
    AND tsOrderID.REGISTRATION_EXPIRY_DATE >= ? -- 参数2: 当前系统时间
ORDER BY 
    taApp.APP_ID ASC;
